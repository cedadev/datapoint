    def __repr__(self):
        """Python representation of standard DataPoint metadata"""
        repr = [str(self)]
        for k, v in self._meta.items():
            repr.append(f' - {k}: {v}')
        return '\n'.join(repr)
    
    def __dict__(self):
        """
        Dictionary Representation for User-facing classes."""
        return self._meta

    @property
    def meta(self):
        """
        Retrieve the ``meta`` values (read-only)
        """
        return self._meta
